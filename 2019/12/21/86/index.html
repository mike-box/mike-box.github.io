5230. 缀点成线

题目

在一个 XY 坐标系中有一些点，我们用数组 coordinates 来分别记录它们的坐标，其中 coordinates[i] = [x, y] 表示横坐标为 x、纵坐标为 y 的点。

请你来判断，这些点是否在该坐标系中属于同一条直线上，是则返回 true，否则请返回 false。

题意

给定点的集合是否在一条直线上

思路

  1. 依次比较相邻三个点构成的斜率是否相等，因为相邻的两个点有公共的交点，如果斜率相等，则认为其在一条直线上。
  2. 由于可能存在斜率无限大的情况，这时我们避免用除法，斜率用(x,y)表示，如果两个斜率相等则可以用等式判断 class Solution {
    public:
    bool checkStraightLine(vector>& coordinates) {
     int n = coordinates.size();         
 int x = coordinates[1][0] - coordinates[0][0]; 
 int y = coordinates[1][1] - coordinates[0][1]; 

    for(int i = 1; i < n; ++i){
        int x1 = coordinates[i][0] - coordinates[i-1][0]; 
        int y1 = coordinates[i][1] - coordinates[i-1][1]; 
        if(y1*x != x1*y){
            return false;
        }
    }

    return true;
}

};

cstruct TrieNode{
bool isfolder;
TrieNode * next[27];
TrieNode(){
this->isfolder = false;
for(int i = 0;i < 27; ++i){
this->next[i] = NULL;
}
}
};

bool insertTrie(TrieNode * root,string word){
if(!root){
return false;
}

TrieNode * node = root;
for(auto c : word){
    if(c == '/'){
        if(!node->next[26]){
            node->next[26] = new TrieNode();
        }
        node = node->next[26]; 
    }else{
        if(!node->next[c-'a']){
            node->next[c-'a'] = new TrieNode();
        }
        node = node->next[c-'a'];
    }
}

node->isfolder = true;
return true;

}

bool isSubFolder(TrieNode * root,string folder){
if(!root){
return false;
}

TrieNode * node = root;
for(int i = 0;i < folder.size(); ++i){
    if(folder[i] == '/'){
        if(!node->next[26]){
            return false;
        }
        node = node->next[26]; 
    }else{
        if(!node->next[folder[i]-'a']){
            return false;
        }
        node = node->next[folder[i]-'a'];
    }
    if(node->isfolder && folder[i+1] == '/'){
        return true;
    }
}

return false;

}

class Solution {
public:
vector removeSubfolders(vector& folder) {
vector res;
sort(folder.begin(),folder.end());

    TrieNode * root = new TrieNode();
    for(auto s : folder){
        if(!isSubFolder(root,s)){
            res.push_back(s);
            insertTrie(root,s);
        }
    }

    return res;
}

};

class Solution {
public:
int balancedString(string s) {
int n = s.size();
int aver = n/4;
int res = n;
unordered_map cnt;
for(int i = 0; i < s.size(); ++i){
cnt[s[i]]++;
}

    bool balance = true;
    for(auto c : cnt){
        if(c.second > aver){
            balance = false;
        }
    }
    if(balance){
        return 0;
    }

    int l = 0;
    int r = 0;
    for(r = 0 ; r < n; ++r){
        cnt[s[r]]--;

        while(l < n && l <= r &&
               cnt['Q'] <= aver&& \
               cnt['W'] <= aver&& \
               cnt['E'] <= aver&& \
               cnt['R'] <= aver){
               res = min(res,r-l+1);
               cnt[s[l]]++;
               l++;           
        }
    }

    return res;
}

};

static bool cmp(vector & a,vector &b){
if(a[0] == b[0]){
return a[2] > b[2];
}

return a[0] > b[0];

}

class Solution {
public:
int jobScheduling(vector& startTime, vector& endTime, vector& profit) {
int n = startTime.size();
int res = 0;
vector> work(n);
map dp;

    for(int i = 0;i < n; ++i){
        work[i].push_back(startTime[i]);
        work[i].push_back(endTime[i]);
        work[i].push_back(profit[i]);
    }
    /*start time sort*/
    sort(work.begin(),work.end(),cmp);
    /*dp*/
    for(int i = 0; i < n; ++i){
        auto it1 = dp.lower_bound(work[i][0]);
        auto it2 = dp.lower_bound(work[i][1]);
        if(it1 != dp.end()){
            dp[work[i][0]] = max(it1->second,work[i][2]);
        }
        if(it2 != dp.end()){
            dp[work[i][0]] = max(dp[work[i][0]],it2->second+work[i][2]);
        }
        dp[work[i][0]] = max(work[i][2],dp[work[i][0]]);
        res = max(res,dp[work[i][0]]);
    }

    return res;
}

};


def jobScheduling(self, startTime, endTime, profit):
jobs = sorted(zip(startTime, endTime, profit), key=lambda v: v[1])
dp = [[0, 0]]
for s, e, p in jobs:
i = bisect.bisect(dp, [s + 1]) - 1
if dp[i][1] + p > dp[-1][1]:
dp.append([e, dp[i][1] + p])
return dp[-1][1]
