5088. 等差数列中缺失的数字

题目

有一个数组，其中的值符合等差数列的数值规律，也就是说：

在 0 <= i < arr.length - 1 的前提下，arr[i+1] - arr[i] 的值都相等。
我们会从该数组中删除一个 既不是第一个 也 不是最后一个的值，得到一个新的数组 arr。

给你这个缺值的数组 arr，请你帮忙找出被删除的那个数。

题意

寻找等差序列中缺失的一个数

思路

  1. 此题比较简单，寻找等差序列中缺少的关键数字；

  2. 先求出等差序列的等差add = (arr[n-1] - arr[0])/n,如果发现当前数字减去序列中的前一个数字与等差不相等，则表明该数字之前缺失一个数，缺失的数字为

    class Solution {
    public:
    int missingNumber(vector& arr) {

     int n = arr.size();
 int add = (arr[n-1] - arr[0])/n;

 for(int i = 1; i < n; ++i){
     if(arr[i] - arr[i-1] != add){
         return arr[i-1] + add;
     }
 }

 return 0;

    }
    };

    class Solution {
    public:
    static bool cmp(vector & a,vector & b){

     return a[0] < b[0];

    }

    vector minAvailableDuration(vector>& slots1, vector>& slots2, int duration) {

     int i = 0;
 int j = 0;

 sort(slots1.begin(),slots1.end(),cmp);
 sort(slots2.begin(),slots2.end(),cmp);

 while(i < slots1.size() && j < slots2.size()){
     int a = max(slots1[i][0],slots2[j][0]);
     int b = min(slots1[i][1],slots2[j][1]);
     if(b - a >= duration){
         return vector<int>({a,a+duration});
     }

     if(b == slots1[i][1]){
         ++i;
     }
     if(b == slots2[j][1]){
         ++j;
     }
 }

 return vector<int>({});

    }
    };

class Solution {
public:
double probabilityOfHeads(vector& prob, int target) {
int n = prob.size();
vector> dp(n+1,vector(n+1,0));

    dp[1][0] = 1 - prob[0];
    dp[1][1] = prob[0];

    for(int i = 2; i <= n; ++i){
        for(int j = 0; j <= i; ++j){
            if(j > 0){
                dp[i][j] += prob[i-1]*dp[i-1][j-1];
            }
            if(i-1 >= j){
                dp[i][j] += (1-prob[i-1])*dp[i-1][j];
            }
        }
    }

    return dp[n][target];
}

};
