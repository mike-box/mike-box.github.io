    dp[i+1][j][k] = min(dp[i+1][j][k],dp[i][l][k] + l_{i + 1} \quad if r \neq k \
    dp[i+1][j][k] = min(dp[i+1][j][k],dp[i][l][k] + l_{i + 1} - 1 \quad if r = k \
